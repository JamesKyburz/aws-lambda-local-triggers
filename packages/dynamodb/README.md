run an aws lambda trigger for dynamodb locally.

### usage

```javascript
const loop = require('aws-lambda-local-dynamodb-trigger')
const { streams } = require('./dynamodb')

loop({
  streamArn: 'x',
  streams,
  async trigger (records) {
  }
})
```
