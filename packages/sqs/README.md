run an aws lambda trigger for sqs locally.

### usage

```javascript
const loop = require('aws-lambda-local-sqs-trigger')
const sqs = require('./sqs')

loop({
  queueUrl: 'http://localhost:9324/queue/test',
  sqs,
  async trigger (event) {
    // event.Records
  }
})
```
